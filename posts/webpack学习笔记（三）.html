# 1.webpack处理图片，常用loader:

# file-loader--引入图片

# 安装：npm install file-loader -s

# 配置：在module的rules里添加：

{
	test:/\.(png|jpg|jpeg|gif)$/,//表示处理的文件类型
	use:[
	{
		loader:'file-loader',
		options:{
			name:'[name].[hash:4].[ext]', //定义打包后的文件名，[name]表示原文件，[ext]表示原扩展名
			outputPath:'assets/images', //定义打包文件存放路径
			publicPath:'imgs'//指定在css里引入的路径
		}
	}
	]
}

# url-loader--file-loader的封装，可以代替file-loader

# 安装：npm install url-loader -s

# 配置：同上，把'file-loader'改成'url-loader'

{
    loader:'url-loader',
    options:{
        //其它配置同上
        limit:5000 //把小于5KB的图片转换成base64码，减少请求
    }
}

# img-loader--引入图片

# 安装：npm install img-loader -s

# 配置：

{
    loader:'img-loader'
}

####使用压缩图片插件压缩png： 在img-loader后添加配置，使用imgagemin-pngquant插件，需安装 imagemin 和 imagemin-pngquant npm install imagemin imagemin-pngquant -s

options:{
    plugins:[
        require('imagemin-pngquant')({
            speed:1 //1-11
        })
    ]
}

# html里引入图片

html引入图片直接使用路径不能解析，需用模板字符串请求资源，webpack自带

# 或使用html-loader:

# 安装：npm install html-loader -s

# 配置：

{
	test:/\.(html|html)$/,
	use:[{
		loader:'html-loader',
		options:{
			attrs:["img:src","img:data-src"] //配置使用data-src和src都能解析img标签下的路径
		}
	}]
}

# 3.使用雪碧图：

# postcss-sprites:

安装：npm install postcss-sprites -s 配置：

{
    loader:'postcss-loader',
    options:{
    	plugins:[
    		require('postcss-sprites')({
    			spritesPath:'./dist/assets/sprites' //打包后雪碧图存放位置
    		})
    	]
    }
}

# webpack-spritesmith:

安装：npm install webpack-spritesmith -s 配置：

new webpackSpritesmith({
    src:{
    	cwd:path.join(__dirname,'src/images'),//定义处理图片的来源
    	glob:'*.jpg'//定义处理的图片类型
    },
    target:{
    	image:path.join(__dirname,'dist/sprites/sprite.png'),//定义雪碧图输出的路径
    	css:path.join(__dirname,'dist/sprites/sprite.css'),//Spritesmith会额外生成一个css,定义生成的css名字，再手动引入
    },
    apiOptions:{
    	cssImageRef:'./sprites/sprite.png' //定义在css时引用的路径，和输出路径保持一致
    }
})

ps:path为node自带的模块，直接引入 const path = require('path')
