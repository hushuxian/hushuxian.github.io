##地理定位

navigator.geolocation.getCurrentPosition(success/*成功的回调函数，获取的位置作为唯一参数传入*/,error/*可选，失败的回调函数，positionError对象作为参数*/,options/*可选，传入对象*/)

判断浏览器是否支持geolocation

if(navigator.geolocation){
	//支持
}else{
	alert("您的浏览器不支持地理位置")
}

navigator.geolocation.getCurrentPosition(function(position){
	console.log(
		position.coords.accuracy,//精确度
		position.coords.altitue,//海拔
		position.coords.longitude,//精度
		position.coords.latitude,//纬度
		position.coords.speed,//行进速度
		position.coords.heading,//行进方向
		position.timestamp//时间戳
	)
},function(err){
	alert(err.code);
	//code:2 表示用户拒绝获取位置的请求
	//code:3 表示尝试获取用户信息失败
	//code:4 表示超时
},
{
  enableHighAcuracy: false, //位置是否精确获取
  timeout: 5000,            //获取位置允许的最长时间
  maximumAge: 1000          //多久更新获取一次位置
})

##localStorage 和 sessionStorage localStorage:长久保存整个网站数据，无过期时间，直到手动删除

window.localStorage.setItem("localData","data2");
//setItem(key,value)存储数据
//getItem(key)获取数据
//removeItem(key)删除数据
//clear()删除所有的数据
//存储对象通过JSON.stringify()转成字符串，获取时通过JSON.parse()把字符串转成对象

sessionStorage:临时保存一个窗口的数据，关闭窗口或标签页后清除

 window.sessionStorage.setItem("sessionData","data1");
 //API与localStorage一样

##离线存储 在页面上指定mainifest属性

<html lang="en" manifest="cache.appcache">

mainifest文件以.appcache扩展名存储

CACHE MANIFEST   
# 2019-09-01 v1.0.0 
CACHE:  #表示需要缓存的文件，使用相对路径
style/index.css   
js/index.js
  
NETWORK:  #不会被缓存的文件
login.php   
  
FALLBACK:   #当页面无法访问时的回退页面
/html5/
