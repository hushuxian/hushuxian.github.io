# 1.webpack环境：

# 使用特定配置文件和特定环境进行打包：webpack --config webpack.common.js --env developoment

module.exports = env =>{ //使用env 接收打包传入的参数
    var common = { //配置项
    }
    return common;
}

# 通过编写不同的配置文件区分不同环境

开发环境：webpack.dev.js 开发环境下需要配置的项：如webpack-dev-server 生产环境：webpack.pro.js 生产环境下需要配置的项：如html压缩 通用配置：webpack.common.js

const pro = require('./webpack.pro.js')
const merge = require('webpack-merge') //使用webpack-merge对配置文件合并，先下载
module.exports = env =>{ 
    var common = { 
    ......
    }
    return merge(common,env == 'production' ? pro : dev);
}

# 在package.json里的scripts下配置自定义命令：

"build":"webpack --config webpack.common.js --env production" //通过npm run build打包 "dev":"webpack --config webpack.common.js --env development" //通过npm run dev

# 通用功能需要不同配置的情况：如css 处理，在生产环境下需要在post-css里添加雪碧图处理，postcss-spirites,

在顶部定义postPlugins变量，初始值设为开发环境和生产环境下需要的插件，

let postPlugins= [require('autoprefixer')()]
postPlugins.concat(env ==='development' ? [] : [require('postcss-sprite')]);

修改postcss-loader 对应的plugins 配置项： plugins:postPlugins

# webpack4 以上可以指定mode进行不同环境下的打包

webpack --mode production/development/none

使用默认配置如生产环境对代码进行压缩等 或直接在配置文件里指定mode,一般在webpack4以上配置文件需直接指定mode,否则打包时有警告信息

mode:'development',//production,none

# 2.webpack-dev-server的使用

webpack-dev-server常用功能： 路径重定向 浏览器显示编译错误 接口代理 热更新

# 安装:

npm install webpack-dev-server --save

# 配置文件下添加devServer

devServer:{
	inline:false,//是否在页面内开启编译状态提示，默认true不开启
	port:9001,//定义服务开启端口
	overlay:true,//报错信息是否以遮罩形式显示到页面，默认不显示
    historyApiFallback:{
        rewrites:[{
            from:/^\/([ -~]+)/, // 表示匹配以/开头的任意一到多个字符
            to:function(context){
            	return './' + context.match[1] + '.html'//context.match为正则和输入字符串匹配的结果数组
            }
        }]
    },//是否开启重定向,可接收对象，设置为ture找不到路径则不跳转，或指定规则
    proxy:{ //使用代理跨域请求
        '/':{ //以'/'开头的地址进行代理转发
            target:'https://www.jianshu.com/',//需要请求接口的ip,在请求处不需要再添加
            changeOrigin:true,//改变origin，默认为true
            pathRewrite:{
            	'^/p':'/p/2e7f3b69e51e'//使用短路径重写接口请求路径,以p开头
            },
            //header:{}//请求头设置
        }
    },
    //hot:true,//开启热更新，不刷新浏览器保持更新
    //ps:热更新和extract-teext-css是冲突的，使用时讲extract-text-css禁用
    hotOnly:true, //只使用热更新,开启后更新文件需要手动刷新,可在入口文件最后添加module.hot.accept()接收改变的代码完成刷新
}

inline设为false显示以下状态条： overlay设为true的报错显示

# package.json的script里配置运行webpack-dev-server

"dev":"webpack-dev-server" //使用npm run dev 代替webpack-dev-server命令，--open 表示自动打开浏览器 --env development 传入环境

# 3.使用source-map:查找打包后的代码对应原文件的位置

（1）安装：npm install source-map --s （2）配置： devtool:'eval-souce-map',//取值默认为eval,构建最快；eval-source-map,错误进行简单调试；cheap-module-source-map:构建慢，重构快，定位到原始代码
