# 1.多入口下代码分割：主业务代码+公共依赖+第三方包+webpack运行代码

配置多个htmlPlugin，每个html单独引入一个入口：

new htmlWebpackPlugin({
    filename:'index1.html',
    template:'./src/index1.html',
    chunks:['app1','common'] //index1引入app1作为入口,同时需要把分割后的代码手动引入
}),
new htmlWebpackPlugin({
    filename:'index2.html',
    template:'./src/index2.html',
    chunks:['app2','common']//index2引入app2作为入口
}),

ps:分割后提取出来的代码需手动引入，如配置了runtimeChunk，也需要单独引入runtime ps:每次打包生成的文件没法自动覆盖，需要使用插件把把每次打包后的文件清除

let {CleanWebpackPlugin} = require('clean-webpack-plugin')
new CleanWebpackPlugin(), //清除dist已打包的内容，引入时需引入单独函数{CleanWebpackPlugin}

# 3.webpack4以上代码分割使用splitChunks

optimization:{
    splitChunks:{
        chunks:'initial',//initial,all,async指定检查的文件，initial表示入口文件，all表示所有文件
        minSize :30000,//最小打包文件，默认大于30kb才单独提取，
        cacheGroups:{ //自定义提取的模块
            common:{
                test:/[\\/]node_modules[\\/]/, //提取所有node_modules里的模块
                name:'common',//指定提取出的文件名,需在htmlPlugin手动指定次文件名
                chunks:'all' //对所有文件进行提取
            },
            default: {
                 test: /[\\/]module[\\/]/,
                 minChunks: 2,//一般为非第三方公共模块
               }
        }
    },
    runtimeChunk:true//提取webpack配置的代码
}

# 4.使用异步模块

在js里使用import()动态引入其他模块，webpack自动打包成成一个模块，使用默认命名打包，如果要指定命名，需要在import里使用魔法注释

let  m1 = import (/*webpackChunkName:"aa"*/'./module/module1.js');//使用import 方法异步加载模块，用注释指定打包后的文件名

直接使用import打包时报错，需使用对应的babel插件@babel/plugin-syntax-dynamic

npm install --save-dev @babel/plugin-syntax-dynamic-import

在babel的options里配置

plugins:["@babel/plugin-syntax-dynamic-import"]

# 5.代码体积控制：压缩和treeshaking

压缩配置：

optimization:{
    minimize：true //开启代码压缩
}

webpack4以上直接设置mode:'production',使用生产模式打包，webpack会自动压缩代码。

treeshaking配置 optimization.minimizer 指定Uglify 或直接设置成生产模式

treeshaking作用

export a = function(){};
export b = function(){};

//通过以上形式导出的模块，在另一个文件里引入后，如果只引入不使用，打包后将不会被打包到文件里。
 
export default {
    a:function(){},
    b:function(){}
}
//这种形式导出的模块则不能使用treeshaking
