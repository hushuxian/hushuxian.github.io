##html5跨文档通讯 父文档html

<div id="div1"></div>
<iframe src="msg.html" frameborder="0" id="frame"></iframe>

父文档js

function sendMsg(text){ //text类型可以是任意数据类型
	var msg = JSON.stringify(text);//如传入对象需要序列化此对象
	var frame = document.getElementById("frame").contentWindow; //获取子文档的contentWindow对象
	setTimeout(function(){
		frame.postMessage(msg,"file:///C:/workspace/vue/h5/msg.html") //通过contentWindow对象的piostMessage发送信息，第一个参数是发送的信息，第二个是发送到的文档，使用file协议的话需要设置浏览器，否则报错
	},1000)
}
sendMsg({say:'hello',from:'hu'});

子文档

//使用message事件监听消息
window.addEventListener("message",function(event){
	document.getElementById("msgContent").innerHTML = "message from: " + event.origin + '\n message content:' +event.data;//event.origin消息的发送者，event.data消息内容
	setTimeout(function(){
		event.source.postMessage("收到",event.origin)//event.source表示消息的接收对象
	},1000)
})
